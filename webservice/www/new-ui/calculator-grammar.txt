--------- Expressions ---------

func-call  : WORD"(" (expr ",")* ")"

atom       : NUMBER | WORD | "(" expr ")" | func-call

signed-atom: ("+"|"-") atom

product    : signed-atom (("*"|"/") product)*

arithmatic : product (("+"|"-") arithmatic)*

var-def    : "var" WORD "=" expr

func-def   : "var()" WORD "=" program

definitions: var-def | func-def

expr       : definitions | arithmatic

------------------------------


program: (expr ";")*
