--------- Expressions ---------

func-call  : WORD"(" (expr ",")* ")"

atom       : NUMBER | WORD | "(" expr ")" | func-call

signed-atom: ("+"|"-") atom

product    : signed-atom (("*"|"/") product)*

arithmatic : product (("+"|"-") arithmatic)*

func-def   : "var()" WORD "=" program "rav"

var-def    : ("var" WORD "=" expr) | func-def

definitions: var-def

expr       : definitions | arithmatic

------------------------------


program: (expr ";")*
