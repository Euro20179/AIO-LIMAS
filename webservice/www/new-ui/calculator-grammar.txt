--------- Expressions ---------

func-call  : WORD"(" (expr ",")* ")"

atom       : NUMBER | WORD | "(" expr ")" | func-call

signed-atom: ("+"|"-") atom

product    : signed-atom (("*"|"/") signed-atom)*

arithmatic : product (("+"|"-") product)*

comparison : arithmatic ((">"|"<"|"=="|"<="|">=") comparison)*

func-def   : "var()" WORD "=" program "rav"

var-def    : ("var" WORD "=" expr) | func-def

expr       : var-def | comparison

------------------------------


program: (expr ";")*
